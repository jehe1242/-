        function drawStampOnContext(targetCtx, targetCanvas, config, isTransparent = false) {

            const { name, scale, weightRaw, spacingRaw, borderW, shape, texture } = config;

            if (!name) return;


            const spacingOffset = (spacingRaw - 50) * 0.8; 


            targetCtx.clearRect(0, 0, targetCanvas.width, targetCanvas.height);

            if (!isTransparent) {

                targetCtx.fillStyle = "white";

                targetCtx.fillRect(0, 0, targetCanvas.width, targetCanvas.height);

            }


            const size = targetCanvas.width * 0.75;

            const x = (targetCanvas.width - size) / 2;

            const y = (targetCanvas.height - size) / 2;

            const centerX = targetCanvas.width / 2;

            const centerY = targetCanvas.height / 2;

            const stampColor = "#b91c1c";


            targetCtx.strokeStyle = stampColor;

            targetCtx.lineWidth = borderW * (targetCanvas.width / 400);

            targetCtx.lineJoin = "miter";


            if (shape === 'square') {

                targetCtx.strokeRect(x, y, size, size);

            } else {

                targetCtx.beginPath();

                targetCtx.arc(centerX, centerY, size/2, 0, Math.PI * 2);

                targetCtx.stroke();

            }


            targetCtx.fillStyle = stampColor;

            targetCtx.textAlign = "center";

            targetCtx.textBaseline = "middle";


            const shadowBlur = weightRaw > 5 ? (weightRaw - 5) * 0.8 : 0;

            if (shadowBlur > 0) {

                targetCtx.shadowColor = stampColor;

                targetCtx.shadowBlur = shadowBlur * (targetCanvas.width / 400);

            } else {

                targetCtx.shadowBlur = 0;

            }


            const chars = name.split('');

            const count = chars.length;

            const offsetMultiplier = shape === 'ellipse' ? 0.9 : 1.0;

            const weightString = weightRaw > 7 ? '800' : (weightRaw > 4 ? '700' : '400');

            const fontSizeBase = targetCanvas.width / 400;


            if (count === 1) {

                targetCtx.font = `${weightString} ${180 * scale * fontSizeBase}px 'Nanum Myeongjo'`;

                targetCtx.fillText(chars[0], centerX, centerY);

            } else if (count === 2) {

                targetCtx.font = `${weightString} ${140 * scale * offsetMultiplier * fontSizeBase}px 'Nanum Myeongjo'`;

                const baseGap = (size/4.5) * scale;

                targetCtx.fillText(chars[0], centerX, centerY - baseGap - (spacingOffset * fontSizeBase));

                targetCtx.fillText(chars[1], centerX, centerY + baseGap + (spacingOffset * fontSizeBase));

            } else if (count === 3) {

                targetCtx.font = `${weightString} ${100 * scale * offsetMultiplier * fontSizeBase}px 'Nanum Myeongjo'`;

                const baseGap = (size/3.2) * scale;

                targetCtx.fillText(chars[0], centerX, centerY - baseGap - (spacingOffset * fontSizeBase));

                targetCtx.fillText(chars[1], centerX, centerY);

                targetCtx.fillText(chars[2], centerX, centerY + baseGap + (spacingOffset * fontSizeBase));

            } else if (count === 4) {

                targetCtx.font = `${weightString} ${110 * scale * offsetMultiplier * fontSizeBase}px 'Nanum Myeongjo'`;

                const baseGap = (size/4) * scale;

                const adjGap = baseGap + (spacingOffset * 0.5 * fontSizeBase);

                targetCtx.fillText(chars[0], centerX + adjGap, centerY - adjGap);

                targetCtx.fillText(chars[1], centerX + adjGap, centerY + adjGap);

                targetCtx.fillText(chars[2], centerX - adjGap, centerY - adjGap);

                targetCtx.fillText(chars[3], centerX - adjGap, centerY + adjGap);

            }


            targetCtx.shadowBlur = 0;


            if (texture === 'rough') {

                applyNoiseOnContext(targetCtx, targetCanvas, x, y, size, stampColor, shape, isTransparent);

            }

        }


        function applyNoiseOnContext(targetCtx, targetCanvas, startX, startY, size, color, shape, isTransparent) {

            const centerX = targetCanvas.width / 2;

            const centerY = targetCanvas.height / 2;

            const radius = size / 2;

            const scaleFactor = targetCanvas.width / 400;


            function isInside(px, py) {

                if (shape === 'square') {

                    return px >= startX && px <= startX + size && py >= startY && py <= startY + size;

                } else {

                    const dx = px - centerX;

                    const dy = py - centerY;

                    return (dx*dx + dy*dy) <= (radius + (10 * scaleFactor)) * (radius + (10 * scaleFactor));

                }

            }


            if (isTransparent) {

                targetCtx.globalCompositeOperation = 'destination-out';

            } else {

                targetCtx.fillStyle = "white";

            }

            

            for (let i = 0; i < 500; i++) {

                const px = (startX - (10 * scaleFactor)) + Math.random() * (size + (20 * scaleFactor));

                const py = (startY - (10 * scaleFactor)) + Math.random() * (size + (20 * scaleFactor));

                if (isInside(px, py)) {

                    const r = Math.random() * 2 * scaleFactor;

                    targetCtx.beginPath();

                    targetCtx.arc(px, py, r, 0, Math.PI * 2);

                    targetCtx.fill();

                }

            }

            

            targetCtx.globalCompositeOperation = 'source-over';

            targetCtx.fillStyle = color;

            for (let i = 0; i < 60; i++) {

                const px = (startX - (10 * scaleFactor)) + Math.random() * (size + (20 * scaleFactor));

                const py = (startY - (10 * scaleFactor)) + Math.random() * (size + (20 * scaleFactor));

                if (isInside(px, py)) {

                    const r = Math.random() * 1.2 * scaleFactor;

                    targetCtx.beginPath();

                    targetCtx.arc(px, py, r, 0, Math.PI * 2);

                    targetCtx.fill();

                }

            }

        }

        function generateStamp() {

            if (!nameInput.value.trim()) {

                initCanvas();

                return;

            }

            updateLabels();

            drawStampOnContext(ctx, canvas, getCurrentConfig(), false);

        }


        function addToHistory() {

            const config = getCurrentConfig();

            if (!config.name) return;


            // 히스토리 리스트에 데이터 추가

            stampHistory.unshift(config);

            renderHistory();

        }


        function renderHistory() {

            historyList.innerHTML = "";

            if (stampHistory.length === 0) {

                historyList.appendChild(emptyMsg);

                return;

            }


            stampHistory.forEach(item => {

                const wrapper = document.createElement('div');

                wrapper.className = "history-item bg-white p-3 rounded-lg border border-gray-100 flex flex-col items-center relative group";

                

                // 작은 캔버스 생성

                const hCanvas = document.createElement('canvas');

                hCanvas.width = 150;

                hCanvas.height = 150;

                hCanvas.className = "w-full h-auto rounded border border-gray-50";

                const hCtx = hCanvas.getContext('2d');

                

                // 해당 컨피그로 그리기

                drawStampOnContext(hCtx, hCanvas, item, false);


                const info = document.createElement('div');

                info.className = "mt-2 text-center";

                info.innerHTML = `<span class="text-xs font-bold text-gray-700">${item.name}</span><br>

                                  <span class="text-[9px] text-gray-400">${item.shape === 'square' ? '사각형' : '타원형'} / 두께:${item.weightRaw}</span>`;


                const actions = document.createElement('div');

                actions.className = "absolute top-1 right-1 opacity-0 group-hover:opacity-100 transition-opacity flex gap-1";

                

                // 삭제 버튼

                const delBtn = document.createElement('button');

                delBtn.innerHTML = "×";

                delBtn.className = "w-5 h-5 bg-red-500 text-white rounded-full flex items-center justify-center text-xs shadow hover:bg-red-600";

                delBtn.onclick = (e) => {

                    e.stopPropagation();

                    stampHistory = stampHistory.filter(h => h.id !== item.id);

                    renderHistory();

                };


                wrapper.onclick = () => {

                    restoreFromConfig(item);

                };


                actions.appendChild(delBtn);

                wrapper.appendChild(hCanvas);

                wrapper.appendChild(info);

                wrapper.appendChild(actions);

                historyList.appendChild(wrapper);

            });

        }


        function restoreFromConfig(config) {

            nameInput.value = config.name;

            fontSizeSlider.value = config.scale * 100;

            fontWeightSlider.value = config.weightRaw;

            spacingSlider.value = config.spacingRaw;

            borderWidthSlider.value = config.borderW;

            setRadioValue('shape', config.shape);

            setRadioValue('texture', config.texture);

            generateStamp();

        }


        function clearCanvas() {

            nameInput.value = "";

            fontSizeSlider.value = 100;

            fontWeightSlider.value = 5;

            spacingSlider.value = 50;

            borderWidthSlider.value = 12;

            updateLabels();

            initCanvas();

        }


        function downloadStamp() {

            const config = getCurrentConfig();

            if (!config.name) return;

            const offCanvas = document.createElement('canvas');

            offCanvas.width = 1000; // 고해상도 다운로드

            offCanvas.height = 1000;

            const offCtx = offCanvas.getContext('2d');

            drawStampOnContext(offCtx, offCanvas, config, true);

            const link = document.createElement('a');

            link.download = `stamp_${config.name}.png`;

            link.href = offCanvas.toDataURL('image/png');

            link.click();

        }


        // Events

        [fontSizeSlider, fontWeightSlider, spacingSlider, borderWidthSlider].forEach(slider => {

            slider.addEventListener('input', generateStamp);

        });


        shapeInputs.forEach(input => input.addEventListener('change', generateStamp));

        textureInputs.forEach(input => input.addEventListener('change', generateStamp));


        nameInput.addEventListener('input', generateStamp);

        

        addToHistoryBtn.addEventListener('click', addToHistory);

        clearBtn.addEventListener('click', clearCanvas);

        downloadBtn.addEventListener('click', downloadStamp);

        clearHistoryBtn.addEventListener('click', () => {

            if(confirm("모든 비교 목록을 삭제할까요?")) {

                stampHistory = [];

                renderHistory();

            }

        });


        window.onload = initCanvas;

    </script>
